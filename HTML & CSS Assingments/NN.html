// Get references to the textarea and buttons const textArea = document.getElementById('text-area'); const clearButton = document.querySelector('button[onclick="clearText()"]'); const saveButton = document.querySelector('button[onclick="saveText()"]'); // Clear the text area function clearText() { if (confirm('Are you sure you want to clear the text?')) { textArea.value = ''; } } // Save the text area content as a .txt file function saveText() { const text = textArea.value; if (text.trim() === '') { alert('There is no text to save!'); return; } const blob = new Blob([text], { type: 'text/plain' }); const link = document.createElement('a'); link.href = URL.createObjectURL(blob); link.download = 'notepad.txt'; link.click(); alert('File saved successfully!'); } // Optionally add keyboard shortcuts for convenience document.addEventListener('keydown', (event) => { if (event.ctrlKey && event.key === 's') { // Ctrl + S to save event.preventDefault(); saveText(); } else if (event.ctrlKey && event.key === 'n') { // Ctrl + N to clear event.preventDefault(); clearText(); } });
